const https = require('https');

function makeRequest(url, method = 'GET', data = null, headers = {}) {
  return new Promise((resolve, reject) => {
    console.log(`üì° –ó–∞–ø—Ä–æ—Å: ${method} ${url}`);
    
    const urlObj = new URL(url);
    const options = {
      hostname: urlObj.hostname,
      path: urlObj.pathname + urlObj.search,
      method: method,
      headers: {
        'Content-Type': 'application/json',
        ...headers
      }
    };

    const req = https.request(options, (res) => {
      let body = '';
      res.on('data', chunk => body += chunk);
      res.on('end', () => {
        try {
          const parsed = JSON.parse(body);
          console.log(`‚úÖ –û—Ç–≤–µ—Ç –ø–æ–ª—É—á–µ–Ω, —Å—Ç–∞—Ç—É—Å: ${res.statusCode}`);
          resolve(parsed);
        } catch (e) {
          console.log(`‚úÖ –û—Ç–≤–µ—Ç –ø–æ–ª—É—á–µ–Ω (—Ç–µ–∫—Å—Ç), —Å—Ç–∞—Ç—É—Å: ${res.statusCode}`);
          resolve(body);
        }
      });
    });

    req.on('error', (err) => {
      console.error(`‚ùå –û—à–∏–±–∫–∞ –∑–∞–ø—Ä–æ—Å–∞: ${err.message}`);
      reject(err);
    });
    
    if (data) req.write(JSON.stringify(data));
    req.end();
  });
}

async function getCdekToken(clientId, clientSecret, isTest) {
  console.log(`üîê –ü–æ–ª—É—á–µ–Ω–∏–µ —Ç–æ–∫–µ–Ω–∞ CDEK (—Ç–µ—Å—Ç: ${isTest})`);
  
  const baseUrl = isTest 
    ? 'https://api.edu.cdek.ru/v2'
    : 'https://api.cdek.ru/v2';

  const params = new URLSearchParams({
    grant_type: 'client_credentials',
    client_id: clientId,
    client_secret: clientSecret
  }).toString();

  try {
    const response = await makeRequest(
      `${baseUrl}/oauth/token?${params}`,
      'POST',
      null,
      { 'Content-Type': 'application/x-www-form-urlencoded' }
    );

    if (response.access_token) {
      console.log(`‚úÖ –¢–æ–∫–µ–Ω –ø–æ–ª—É—á–µ–Ω`);
      return response.access_token;
    } else {
      throw new Error('No access_token in response');
    }
  } catch (error) {
    console.error(`‚ùå –û—à–∏–±–∫–∞ –ø–æ–ª—É—á–µ–Ω–∏—è —Ç–æ–∫–µ–Ω–∞: ${error.message}`);
    throw error;
  }
}

exports.handler = async (event) => {
  console.log('üöÄ –ù–∞—á–∞–ª–æ –æ–±—Ä–∞–±–æ—Ç–∫–∏ –∑–∞–ø—Ä–æ—Å–∞ search-cities-cdek');
  console.log('üìù Event:', JSON.stringify(event, null, 2));
  
  try {
    const clientId = process.env.CDEK_CLIENT_ID;
    const clientSecret = process.env.CDEK_CLIENT_SECRET;
    const isTest = process.env.CDEK_TEST_MODE === 'true';

    console.log(`‚öôÔ∏è –ö–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è: —Ç–µ—Å—Ç=${isTest}, clientId=${clientId ? '—É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω' : '–ù–ï —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω'}`);

    if (!clientId || !clientSecret) {
      console.error('‚ùå –û—Ç—Å—É—Ç—Å—Ç–≤—É—é—Ç CDEK kredencials');
      return {
        statusCode: 500,
        headers: {
          'Access-Control-Allow-Origin': '*',
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ error: 'CDEK API credentials not configured' }),
      };
    }

    // –ü—ã—Ç–∞–µ–º—Å—è –ø–æ–ª—É—á–∏—Ç—å –ø–∞—Ä–∞–º–µ—Ç—Ä 'q' —Ä–∞–∑–Ω—ã–º–∏ —Å–ø–æ—Å–æ–±–∞–º–∏
    let query = '';
    
    // –°–ø–æ—Å–æ–± 1: queryStringParameters (—Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω—ã–π)
    if (event.queryStringParameters?.q) {
      query = event.queryStringParameters.q;
      console.log(`üìç –ü–∞—Ä–∞–º–µ—Ç—Ä –Ω–∞–π–¥–µ–Ω –≤ queryStringParameters: "${query}"`);
    }
    // –°–ø–æ—Å–æ–± 2: multiValueQueryStringParameters
    else if (event.multiValueQueryStringParameters?.q?.[0]) {
      query = event.multiValueQueryStringParameters.q[0];
      console.log(`üìç –ü–∞—Ä–∞–º–µ—Ç—Ä –Ω–∞–π–¥–µ–Ω –≤ multiValueQueryStringParameters: "${query}"`);
    }
    // –°–ø–æ—Å–æ–± 3: –∏–∑ path (–µ—Å–ª–∏ —á—Ç–æ-—Ç–æ –ø–æ—à–ª–æ –Ω–µ —Ç–∞–∫)
    else if (event.rawQueryString) {
      const params = new URLSearchParams(event.rawQueryString);
      query = params.get('q') || '';
      console.log(`üìç –ü–∞—Ä–∞–º–µ—Ç—Ä –∏–∑–≤–ª–µ—á–µ–Ω –∏–∑ rawQueryString: "${query}"`);
    }

    query = query.trim();
    console.log(`üîç –§–∏–Ω–∞–ª—å–Ω—ã–π –ø–æ–∏—Å–∫–æ–≤—ã–π –∑–∞–ø—Ä–æ—Å: "${query}"`);

    if (query.length < 2) {
      console.warn(`‚ö†Ô∏è –ó–∞–ø—Ä–æ—Å —Å–ª–∏—à–∫–æ–º –∫–æ—Ä–æ—Ç–∫–∏–π (${query.length} —Å–∏–º–≤–æ–ª–æ–≤)`);
      return {
        statusCode: 400,
        headers: {
          'Access-Control-Allow-Origin': '*',
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ error: 'Search query must be at least 2 characters' }),
      };
    }

    const baseUrl = isTest ? 'https://api.edu.cdek.ru/v2' : 'https://api.cdek.ru/v2';
    
    console.log(`üîë –ü–æ–ª—É—á–µ–Ω–∏–µ —Ç–æ–∫–µ–Ω–∞ –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –≥–æ—Ä–æ–¥–æ–≤...`);
    const token = await getCdekToken(clientId, clientSecret, isTest);

    console.log(`üìã –ó–∞–ø—Ä–æ—Å —Å–ø–∏—Å–∫–∞ –≤—Å–µ—Ö –≥–æ—Ä–æ–¥–æ–≤ –∏–∑ CDEK...`);
    const citiesResponse = await makeRequest(
      `${baseUrl}/location/cities`,
      'GET',
      null,
      { 'Authorization': `Bearer ${token}` }
    );

    console.log(`üì• –ü–æ–ª—É—á–µ–Ω–æ –æ—Ç–≤–µ—Ç–∞, —Ç–∏–ø: ${Array.isArray(citiesResponse) ? '–º–∞—Å—Å–∏–≤' : '–æ–±—ä–µ–∫—Ç'}`);

    const citiesData = Array.isArray(citiesResponse) 
      ? citiesResponse 
      : (citiesResponse.data && Array.isArray(citiesResponse.data))
        ? citiesResponse.data
        : [];

    console.log(`üèôÔ∏è –í—Å–µ–≥–æ –≥–æ—Ä–æ–¥–æ–≤ –≤ –ë–î CDEK: ${citiesData.length}`);

    // –§–∏–ª—å—Ç—Ä—É–µ–º –ø–æ –ø–æ–∏—Å–∫–æ–≤–æ–º—É –∑–∞–ø—Ä–æ—Å—É
    const searchLower = query.toLowerCase();
    const filtered = citiesData
      .filter(city => {
        const cityName = (city.city || '').toLowerCase();
        return cityName.includes(searchLower);
      })
      .slice(0, 50)
      .map(city => ({
        code: city.city_code,
        name: city.city,
        region: city.region
      }));

    console.log(`‚úÖ –ù–∞–π–¥–µ–Ω–æ –≥–æ—Ä–æ–¥–æ–≤: ${filtered.length}`);
    console.log(`üìä –†–µ–∑—É–ª—å—Ç–∞—Ç—ã: ${JSON.stringify(filtered.slice(0, 3))}`);

    return {
      statusCode: 200,
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, OPTIONS',
        'Access-Control-Allow-Headers': 'Content-Type',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        success: true,
        data: filtered
      }),
    };

  } catch (error) {
    console.error(`üí• –ö—Ä–∏—Ç–∏—á–µ—Å–∫–∞—è –æ—à–∏–±–∫–∞: ${error.message}`);
    console.error(`üìç Stack: ${error.stack}`);
    
    return {
      statusCode: 500,
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ 
        error: error.message || 'Internal error',
        details: error.toString()
      }),
    };
  }
};
